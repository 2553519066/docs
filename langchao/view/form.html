<Form />

组件继承了 element-plus form 所有属性, 并扩展了更多属性

<Form v-model="demoForm" :props="formProps" :loading="false" style="margin-bottom: 20px">
  <template #form-address>字段插槽</template>
</Form>

<!-- 表单项单独使用 -->
<FormItem
  v-model="formItem"
  :items="[
    {
      prop: 'formItem',
      label: '',
      type: 'selectRemote',
      remoteUrl: '/api',
      remoteParams: {},
      remoteFilterKey: 'keyword',
      remoteInit: true,
      remoteValue: 'code',
      ...
    },
  ]"
></FormItem>

<script setup>
import { ref } from "vue"
const formProps = [{ prop: "address", label: "地址", type: "text", disabled: false }]
const demoForm = ref({})
</script>
